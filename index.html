'bold' : 'normal'); if (opts.size) cell.setFontSize(opts.size); if (opts.halign) cell.setHorizontalAlignment(opts.halign); if (opts.valign) cell.setVerticalAlignment(opts.valign); if (opts.wrap !== undefined) cell.setWrap(opts.wrap); if (opts.border) cell.setBorder(true,true,true,true,true,true,'#BFBFBF', SpreadsheetApp.BorderStyle.SOLID); } return cell; } // ── 헤더 생성 (1~8행) ── function buildHeader(sheet, jiraTitle) { // ROW 1: 빈 행 sheet.setRowHeight(1, 10); var countFormulas = [ ['Passed','N/E'], ['Con-Passed','Rejected'], ['Failed','Blocked'], ['Reopen','Deleted'], ]; POC_LIST.forEach(function(poc) { var sc = poc.startCol; var scL = colLetter(sc); // Status 열 문자 (K, O, S...) countRows.forEach(function(cr, ri) { // 왼쪽 라벨 var lc = sheet.getRange(cr.row, sc); }); sheet.getRange(5,10).setBackground(C.META_BG); // ROW 6: 진행률 AI2Max/Smart3Mini/UHD4 sheet.setRowHeight(6, 15); var pocCols6 = [[7,27],[8,31],[9,35]]; pocCols6.forEach(function(item) { var sc = item[1]; }); // ROW 7~8: TC 헤더 sheet.setRowHeight(7, 13); sheet.setRowHeight(8, 13); var tcHeaders = [ [2,'TC ID'], [3,' JIRA 번호'], [4,'대분류'], [5,'중분류'], [6,'소분류'], [7,'시험 항목'], [8,'Pre-Condition/\nInput Data'], [9,'시험 절차'], [10,'주요 점검 항목'] ]; }); // 행 고정 sheet.setFrozenRows(8); sheet.setFrozenColumns(2); } // ── TC 데이터 행 작성 ── function writeTC(sheet, tcData) { var dv = SpreadsheetApp.newDataValidation() .requireValueInList(['Passed','Con-Passed','Failed','Reopen','N/E','Blocked(N/A)','Rejected','Deleted'], true) .build(); '#FFFFFF' : '#F2F7FB'; // TC 정보 (B~J) var vals = [ [2, tc.tc_id], [3, tc.jira || ''], [4, tc.category_large || ''], [5, tc.category_mid || ''], [6, tc.category_small || ''], [7, tc.test_item || ''], [8, tc.precondition || ''], [9, tc.test_steps || ''], [10,tc.expected_result|| ''], ]; else if (item[0] === 3) cell.setFontColor('#7030A0').setHorizontalAlignment('left'); else if (item[0] <= 6) cell.setHorizontalAlignment('center'); else cell.setHorizontalAlignment('left'); }); // PoC Status + 비고 POC_LIST.forEach(function(poc) { var sc = poc.startCol; // 비고 병합 sheet.getRange(row, sc+1, 1, 3).merge() .setBackground(bg).setFontSize(9).setVerticalAlignment('top'); }); sheet.setRowHeight(row, 26); }); } // ── 메인 doPost ── function doPost(e) { try { var payload = JSON.parse(e.postData.contents); // 웹앱 실행 시 getActiveSpreadsheet() 불가 → openById 사용 var SHEET_ID = '1pl8pGt944mhtWa0TzbsFrdru5Eo3EeB3M8H15KjJJck'; var ss = SpreadsheetApp.openById(SHEET_ID); // 동일 이름 시트 처리 var finalName = sheetName;
